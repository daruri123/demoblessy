#include <stdio.h>

// Function to print the Fibonacci series up to 'n' terms
void printFibonacci(int n) {
    // Initialize the first two terms
    int first = 0, second = 1, next;

    // Print the first term
    printf("Fibonacci series up to %d terms:\n", n);

    // Special case for 0 terms
    if (n <= 0) {
        printf("No terms to display.\n");
        return;
    }

    // Print the first term
    printf("%d ", first);

    // If only one term is required
    if (n == 1) {
        return;
    }

    // Print the second term
    printf("%d ", second);

    // Generate and print the remaining terms
    for (int i = 2; i < n; i++) {
        next = first + second; // Calculate the next term
        printf("%d ", next);   // Print the next term
        first = second;        // Update 'first' to the previous 'second'
        second = next;         // Update 'second' to the new 'next'
    }
    
    printf("\n"); // Print a newline at the end
}

int main() {
    int num_terms;

    // Prompt the user for the number of terms
    printf("Enter the number of terms for Fibonacci series: ");
    scanf("%d", &num_terms);

    // Call the function to print Fibonacci series
    printFibonacci(num_terms);

    return 0;
}
